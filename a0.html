📚 HTML基础

HTML是网页的骨架，定义了内容的结构和语义。

<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <title>页面标题</title>
</head>
<body>
    <h1>主标题</h1>
    <p>段落内容</p>
</body>
</html>
🛠️ HTML标签

HTML常用标签聚合

<!-- 文档结构和元信息 -->
<!DOCTYPE html> <!-- 文档类型声明 -->
<html>        <!-- 根标签 -->
<head>        <!-- 文档头部 -->
    <meta charset="UTF-8"> <!-- 字符编码声明 -->
    <title>网页标题</title>
    <link rel="stylesheet" href="style.css"> <!-- 外部样式表 -->
    <script src="script.js"></script>        <!-- 外部脚本 -->
</head>
<body>        <!-- 网页主体内容 -->

<!-- 结构与布局 -->
<header> <!-- 页眉区域 -->
    <h1>网站标题</h1>
</header>
<nav> <!-- 导航区域 -->
    <ul>
        <li><a href="#home">首页</a></li>
        <li><a href="#about">关于</a></li>
    </ul>
</nav>
<main> <!-- 主要内容区 -->
    <article> <!-- 独立内容块，如文章 -->
        <h2>文章标题</h2>
        <p>文章内容...</p>
    </article>
    <section> <!-- 内容分区 -->
        <h3>分区标题</h3>
        <p>分区内容...</p>
    </section>
    <aside> <!-- 侧边栏内容 -->
        <p>侧边信息</p>
    </aside>
</main>
<footer> <!-- 页脚区域 -->
    <p>版权所有 © 2025</p>
</footer>

<!-- 标题与段落 -->
<h1>主标题</h1>
<h2>二级标题</h2>
<h3>三级标题</h3>
<h4>四级标题</h4>
<h5>五级标题</h5>
<h6>六级标题</h6>
<p>段落文本内容。</p>
